-- whole process --
EXP -> TERM EXP'
EXP' -> ADDOP TERM EXP' | &
ADDOP -> + | -
TERM -> FACTOR TERM'
TERM' -> MULOP FACTOR TERM' | &
MULOP -> *
FACTOR -> ( EXP ) | number
